const axios = require('axios');

// Define your StreamElements API key and endpoint URL
const apiKey = 'YOUR_API_KEY';
const endpointUrl = 'YOUR_API_ENDPOINT_URL';

// Define custom messages for users
const customMessages = {
    'user1': 'Welcome back, {user}! Thanks for lurking!',
    'user2': 'Hey there, {user}! Enjoy your lurk!',
    'user3': 'Nice to see you lurking, {user}! Keep it up!',
    // Add more users and their custom messages as needed
};

// Function to handle custom lurk commands
const handleLurkCommand = async (username) => {
    const customMessage = customMessages[username.toLowerCase()] || 'Shh, someone\'s lurking...';
    const message = customMessage.replace('{user}', username);

    try {
        const response = await axios.post(endpointUrl, {
            apiKey: apiKey,
            command: '!lurk',
            response: message
        });
        console.log(response.data);
    } catch (error) {
        console.error(error);
    }
};

// Example: Handle lurk command for a specific user
handleLurkCommand('user1');

In this code snippet:



Replace 'YOUR_API_KEY' with your actual StreamElements API key.

Replace 'YOUR_API_ENDPOINT_URL' with the correct API endpoint URL provided by StreamElements.

Customize the customMessages object with the desired usernames and messages.

The handleLurkCommand function sends a POST request to the StreamElements API to set up the custom lurk command for a specific user.
